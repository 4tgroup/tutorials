RedisEasy Module

To wrap the set/add, get, del Redis command to one input object parameter. This module executes the command via ioredis

1. Usage

Using in Node.js

var redis = require('./dbAPI-server').RedisEasy;
                         
redis.set(input, function (output) {   //Can replace .set by .add, .get or .del

    if(ouput.status == 0){
        //Handle successful results
        var rs = output.results;
    }else {
        //Handle error
        var errMsg = output.message;
    }
});

add command will execute if the variable of set command do not exist in database and vice versa.

2. Input

The JSON object with the format:

params = {
    "requestID": "REQUEST_ID_TOKEN",
    "metadata": {
        "dbinfo": {
            "DBMS": "redis",
            "cFile": "CONFIG_FILE_PATCH",
            "config": {
                "pin": "",
                "protocol": "",
                "host": "",
                "port": "",
                "user": "",
                "password": "",
                "database": ""
                },
            "connectionString":"CONNECTION_STRING" 
        }
    },
    "data": {
        "variable": "name" or [array],
        "value": "String value" or [array] or {hash},
        "index": [],
        "other": []
        }
   }

3. Example

Set command:

  • String datatype

    • Set the variable "var":"Hello" and will expire in 10 seconds.

      params = {
    "requestID": "REQUEST_ID_TOKEN",
    "metadata": {
        "dbinfo": {
            "DBMS": "redis",
            "cFile": "../config/dbinfo.redis.json"
        }
    },
    "data": {
        "variable": "var",
        "value": "Hello",
        "other": ["EX", 10]
        }
   }

  • Hash datatype

    • Insert the hash object {"hkey1":"hval1", ...} to the variable hash.

      params = {
    "requestID": "REQUEST_ID_TOKEN",
    "metadata": {...},
    "data": {
        "variable": "hash",
        "value": {"hkey1":"val1", "hkey2":"val2", "hkey3":"val3"}
        }
   }

Add command:

  • List datatype

    • Insert the (repeating and unsorted) array ["lval1", "lval2", "lval2"] to the head of the variable list.

      params = {
    "requestID": "REQUEST_ID_TOKEN",
    "metadata": {...},
    "data": {
        "variable": "list",
        "value": ["lval1", "lval2", "lval2"],
        "index": [0],
        }
   }

  • Set datatype

    • Insert the (none-repeating and unsorted) array ["sval1", "sval2", "sval3"] to the variable set.

      params = {
    "requestID": "REQUEST_ID_TOKEN",
    "metadata": {...},
    "data": {
        "variable": "set",
        "value": ["sval1", "sval2", "sval3"]
        }
   }

Get command:

  • String datatype

    • Get the value of the variable var.

      params = {
    "requestID": "REQUEST_ID_TOKEN",
    "metadata": {...},
    "data": {
        "variable": "var"
        }
   }
  • List datatype

    • Get values of the (repeating & unsorted) array from the variable list with the position range from 2 to 10.

      params = {
    "requestID": "REQUEST_ID_TOKEN",
    "metadata": {...},
    "data": {
        "variable": "list",
        "index": [2, 10]
        }
   }

Del command:

  • Set datatype

    • Delete the values "sval1" and "sval3" from the (non-repeating & unsorted) set array.

      params = {
    "requestID": "REQUEST_ID_TOKEN",
    "metadata": {...},
    "data": {
        "variable": "set",
        "value": ["sval1", "sval3"]
        }

  • Hash datatype

    • Delete the values of hash key "hkey1" and "hkey2" from hash object hash.

      params = {
    "requestID": "REQUEST_ID_TOKEN",
    "metadata": {...},
    "data": {
        "variable": "hash",
        "index": ["hkey1", "hkey2"]
        }
   }
